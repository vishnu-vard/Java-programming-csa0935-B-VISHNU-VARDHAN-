import java.util.*;
class Prime
{
public static void main(String arg[])
{
Scanner sc=new Scanner(System.in);
int a,i,count=0;
System.out.println("enter a value:");
a=sc.nextInt();
for(i=1;i<=a;i++)
{
if(a%i==0)
{
count=count+1;
}
}
if(count==2)
{
System.out.println("prime number "+a);
}
else
{
System.out.println("not prime number is "+a);
}
}
}
