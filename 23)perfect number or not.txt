import java.util.*;
class Perfect
{
public static void main(String arg[])
{
Scanner sc=new Scanner(System.in);
System.out.println("enter number:");
int n=sc.nextInt();
int count=0;
for(int i=1;i<n;i++)
{
if(n%i==0)
{
count=count+i;
}
}
if(n==count)
{
System.out.println(n+"is a perfect number:");
}
}
}
