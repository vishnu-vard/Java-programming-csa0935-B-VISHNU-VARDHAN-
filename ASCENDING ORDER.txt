import java.util.*;
class Sorting
{
	public static void main(String arg[])
	{
		int i,j=0;
		int arr[]={0,9,8,7,6,5,4,3,2,1};
		int n=arr.length;
		int temp=0;
		for(i=0;i<n-1;i++)
		{
			for(j=0;j<n-i-1;j++)
			{
				if(arr[j]>arr[j+1])
				{
					temp=arr[j];
					arr[j]=arr[j+1];
					arr[j+1]=temp;
				}
			}
		}
		for(i=0;i<n;i++)
		{
		for(j=0;j<n;j++)
		{
		System.out.print(arr[j]+ " ");
		
		}
		break;
		}
		}
		}
